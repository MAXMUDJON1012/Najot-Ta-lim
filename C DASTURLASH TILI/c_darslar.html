Misol — kalkulyator

kalk.c
#include <stdio.h>

int main() {
    int a = 10, b = 3;
    char op;
    printf("Amal (+,-,*,/): ");
    scanf(" %c", &op);

    switch (op) {
        case '+':
            printf("Natija: %d\n", a + b);
            break;
        case '-':
            printf("Natija: %d\n", a - b);
            break;
        case '*':
            printf("Natija: %d\n", a * b);
            break;
        case '/':
            printf("Natija: %d\n", a / b);
            break;
        default:
            printf("Noto'g'ri amal!\n");
    }
    return 0;
}
Natija (* tanlanganda)
Amal (+,-,*,/): *
Natija: 30
⚠️ break unutilsa — keyingi barcha case'lar ham ishlaydi! Bu "fall-through" deyiladi. Har doim break qo'ying.

Misol — menyu

menyu.c
#include <stdio.h>

int main() {
    int tanlov;

    do {
        printf("\n=== MENYU ===\n");
        printf("1. Salom\n");
        printf("2. Xayr\n");
        printf("0. Chiqish\n");
        printf("Tanlang: ");
        scanf("%d", &tanlov);

        switch (tanlov) {
            case 1: printf("Salom!\n");     break;
            case 2: printf("Xayr!\n");      break;
            case 0: printf("Chiqildi.\n"); break;
            default: printf("Xato!\n");
        }
    } while (tanlov != 0);   // 0 kiritilmasa takrorla

    return 0;
}
